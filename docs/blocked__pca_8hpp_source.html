93template<typename Index_, class EigenVector_>
94struct BlockingDetails {
95 std::vector<Index_> block_size;
96
97 bool weighted = false;
98 typedef typename EigenVector_::Scalar Weight;
99
100 // The below should only be used if weighted = true.
101 std::vector<Weight> per_element_weight;
102 Weight total_block_weight = 0;
103 EigenVector_ expanded_weights;
104};
105
106template<class EigenVector_, typename Index_, typename Block_>
107BlockingDetails<Index_, EigenVector_> compute_blocking_details(
108 Index_ ncells,
109 const Block_* block,
110 scran_blocks::WeightPolicy block_weight_policy,
111 const scran_blocks::VariableWeightParameters& variable_block_weight_parameters)
112{
113 BlockingDetails<Index_, EigenVector_> output;
114 output.block_size = tatami_stats::tabulate_groups(block, ncells);
115 if (block_weight_policy == scran_blocks::WeightPolicy::NONE) {
116 return output;
117 }
118
119 const auto& block_size = output.block_size;
120 size_t nblocks = block_size.size();
121 output.weighted = true;
122 auto& total_weight = output.total_block_weight;
123 auto& element_weight = output.per_element_weight;
124 element_weight.resize(nblocks);
125
126 for (size_t i = 0; i < nblocks; ++i) {
127 auto bsize = block_size[i];
128
129 // Computing effective block weights that also incorporate division by the
130 // block size. This avoids having to do the division by block size in the
131 // 'compute_blockwise_mean_and_variance*()' functions.
132 if (bsize) {
133 typename EigenVector_::Scalar block_weight = 1;
134 if (block_weight_policy == scran_blocks::WeightPolicy::VARIABLE) {
135 block_weight = scran_blocks::compute_variable_weight(bsize, variable_block_weight_parameters);
136 }
137
138 element_weight[i] = block_weight / bsize;
139 total_weight += block_weight;
140 } else {
141 element_weight[i] = 0;
142 }
143 }
144
145 // Setting a placeholder value to avoid problems with division by zero.
146 if (total_weight == 0) {
147 total_weight = 1;
148 }
149
150 // Expanding them for multiplication in the IRLBA wrappers.
151 auto sqrt_weights = element_weight;
152 for (auto& s : sqrt_weights) {
153 s = std::sqrt(s);
154 }
155
156 auto& expanded = output.expanded_weights;
157 expanded.resize(ncells);
158 for (Index_ i = 0; i < ncells; ++i) {
159 expanded.coeffRef(i) = sqrt_weights[block[i]];
160 }
161
162 return output;
163}

169template<typename Num_, typename Value_, typename Index_, typename Block_, typename EigenVector_, typename Float_>
170void compute_sparse_mean_and_variance_blocked(
171 Num_ num_nonzero,
172 const Value_* values,
173 const Index_* indices,
174 const Block_* block,
175 const BlockingDetails<Index_, EigenVector_>& block_details,
176 Float_* centers,
177 Float_& variance,
178 std::vector<Index_>& block_copy,
179 Num_ num_all)
180{
181 const auto& block_size = block_details.block_size;
182 size_t nblocks = block_size.size();
183
184 std::fill_n(centers, nblocks, 0);
185 for (Num_ i = 0; i < num_nonzero; ++i) {
186 centers[block[indices[i]]] += values[i];
187 }
188 for (size_t b = 0; b < nblocks; ++b) {
189 auto bsize = block_size[b];
190 if (bsize) {
191 centers[b] /= bsize;
192 }
193 }
194
195 // Computing the variance from the sum of squared differences.
196 // This is technically not the correct variance estimate if we
197 // were to consider the loss of residual d.f. from estimating
198 // the block means, but it's what the PCA sees, so whatever.
199 variance = 0;
200 std::copy(block_size.begin(), block_size.end(), block_copy.begin());
201
202 if (block_details.weighted) {
203 for (Num_ i = 0; i < num_nonzero; ++i) {
204 Block_ curb = block[indices[i]];
205 auto diff = values[i] - centers[curb];
206 variance += diff * diff * block_details.per_element_weight[curb];
207 --block_copy[curb];
208 }
209 for (size_t b = 0; b < nblocks; ++b) {
210 auto val = centers[b];
211 variance += val * val * block_copy[b] * block_details.per_element_weight[b];
212 }
213 } else {
214 for (Num_ i = 0; i < num_nonzero; ++i) {
215 Block_ curb = block[indices[i]];
216 auto diff = values[i] - centers[curb];
217 variance += diff * diff;
218 --block_copy[curb];
219 }
220 for (size_t b = 0; b < nblocks; ++b) {
221 auto val = centers[b];
222 variance += val * val * block_copy[b];
223 }
224 }
225
226 // COMMENT ON DENOMINATOR:
227 // If we're not dealing with weights, we compute the actual sample
228 // variance for easy interpretation (and to match up with the
229 // per-PC calculations in internal::clean_up).
230 //
231 // If we're dealing with weights, the concept of the sample variance
232 // becomes somewhat weird, but we just use the same denominator for
233 // consistency in clean_up_projected. Magnitude doesn't matter when
234 // scaling for internal::process_scale_vector anyway.
235 variance /= num_all - 1;
236}

239void compute_blockwise_mean_and_variance_realized_sparse(
240 const IrlbaSparseMatrix_& emat, // this should be column-major with genes in the columns.
241 const Block_* block,

244 EigenVector_& variances,

275template<typename Num_, typename Value_, typename Block_, typename Index_, typename EigenVector_, typename Float_>
276void compute_dense_mean_and_variance_blocked(
277 Num_ number,
278 const Value_* values,
279 const Block_* block,
280 const BlockingDetails<Index_, EigenVector_>& block_details,
281 Float_* centers,
282 Float_& variance)
283{
284 const auto& block_size = block_details.block_size;
285 size_t nblocks = block_size.size();
286 std::fill_n(centers, nblocks, 0);
287 for (Num_ r = 0; r < number; ++r) {
288 centers[block[r]] += values[r];
289 }
290 for (size_t b = 0; b < nblocks; ++b) {
291 const auto& bsize = block_size[b];
292 if (bsize) {
293 centers[b] /= bsize;
294 }
295 }
296
297 variance = 0;
298
299 if (block_details.weighted) {
300 for (Num_ r = 0; r < number; ++r) {
301 auto curb = block[r];
302 auto delta = values[r] - centers[curb];
303 variance += delta * delta * block_details.per_element_weight[curb];
304 }
305 } else {
306 for (Num_ r = 0; r < number; ++r) {
307 auto curb = block[r];
308 auto delta = values[r] - centers[curb];
309 variance += delta * delta;
310 }
311 }
312
313 variance /= number - 1; // See COMMENT ON DENOMINATOR above.
314}

340template<typename Value_, typename Index_, typename Block_, class EigenMatrix_, class EigenVector_>
341void compute_blockwise_mean_and_variance_tatami(
342 const tatami::Matrix<Value_, Index_>& mat, // this should have genes in the rows!
343 const Block_* block,
344 const BlockingDetails<Index_, EigenVector_>& block_details,
345 EigenMatrix_& centers,
346 EigenVector_& variances,
347 int nthreads)
348{
349 const auto& block_size = block_details.block_size;
350 size_t nblocks = block_size.size();
351 Index_ ngenes = mat.nrow();
352 Index_ ncells = mat.ncol();
353
354 if (mat.prefer_rows()) {
355 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
356 static_assert(!EigenMatrix_::IsRowMajor);
357 auto mptr = centers.data() + static_cast<size_t>(start) * nblocks; // cast to avoid overflow.
358 std::vector<Index_> block_copy(nblocks);
359
360 std::vector<Value_> vbuffer(ncells);
361
362 if (mat.is_sparse()) {
363 std::vector<Index_> ibuffer(ncells);
364 auto ext = tatami::consecutive_extractor<true>(&mat, true, start, length);
365 for (Index_ r = start, end = start + length; r < end; ++r, mptr += nblocks) {
366 auto range = ext->fetch(vbuffer.data(), ibuffer.data());
367 compute_sparse_mean_and_variance_blocked(range.number, range.value, range.index, block, block_details, mptr, variances[r], block_copy, ncells);
368 }
369 } else {
370 auto ext = tatami::consecutive_extractor<false>(&mat, true, start, length);
371 for (Index_ r = start, end = start + length; r < end; ++r, mptr += nblocks) {
372 auto ptr = ext->fetch(vbuffer.data());
373 compute_dense_mean_and_variance_blocked(ncells, ptr, block, block_details, mptr, variances[r]);
374 }
375 }
376 }, ngenes, nthreads);
377
378 } else {
379 typedef typename EigenVector_::Scalar Scalar;
380
381 std::vector<std::pair<size_t, Scalar> > block_multipliers;
382 block_multipliers.reserve(nblocks);
383 for (size_t b = 0; b < nblocks; ++b) {
384 auto bsize = block_size[b];
385 if (bsize > 1) { // skipping blocks with NaN variances.
386 Scalar mult = bsize - 1; // need to convert variances back into sum of squared differences.
387 if (block_details.weighted) {
388 mult *= block_details.per_element_weight[b];
389 }
390 block_multipliers.emplace_back(b, mult);
391 }
392 }
393
394 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
395 std::vector<std::vector<Scalar> > re_centers, re_variances;
396 re_centers.reserve(nblocks);
397 re_variances.reserve(nblocks);
398 for (size_t b = 0; b < nblocks; ++b) {
399 re_centers.emplace_back(length);
400 re_variances.emplace_back(length);
401 }
402
403 std::vector<Value_> vbuffer(length);
404
405 if (mat.is_sparse()) {
406 std::vector<tatami_stats::variances::RunningSparse<Scalar, Value_, Index_> > running;
407 running.reserve(nblocks);
408 for (size_t b = 0; b < nblocks; ++b) {
409 running.emplace_back(length, re_centers[b].data(), re_variances[b].data(), /* skip_nan = */ false, /* subtract = */ start);
410 }
411
412 std::vector<Index_> ibuffer(length);
413 auto ext = tatami::consecutive_extractor<true>(&mat, false, static_cast<Index_>(0), ncells, start, length);
414 for (Index_ c = 0; c < ncells; ++c) {
415 auto range = ext->fetch(vbuffer.data(), ibuffer.data());
416 running[block[c]].add(range.value, range.index, range.number);
417 }
418
419 for (size_t b = 0; b < nblocks; ++b) {
420 running[b].finish();
421 }
422
423 } else {
424 std::vector<tatami_stats::variances::RunningDense<Scalar, Value_, Index_> > running;
425 running.reserve(nblocks);
426 for (size_t b = 0; b < nblocks; ++b) {
427 running.emplace_back(length, re_centers[b].data(), re_variances[b].data(), /* skip_nan = */ false);
428 }
429
430 auto ext = tatami::consecutive_extractor<false>(&mat, false, static_cast<Index_>(0), ncells, start, length);
431 for (Index_ c = 0; c < ncells; ++c) {
432 auto ptr = ext->fetch(vbuffer.data());
433 running[block[c]].add(ptr);
434 }
435
436 for (size_t b = 0; b < nblocks; ++b) {
437 running[b].finish();
438 }
439 }
440
441 static_assert(!EigenMatrix_::IsRowMajor);
442 auto mptr = centers.data() + static_cast<size_t>(start) * nblocks; // cast to avoid overflow.
443 for (Index_ r = 0; r < length; ++r, mptr += nblocks) {
444 for (size_t b = 0; b < nblocks; ++b) {
445 mptr[b] = re_centers[b][r];
446 }
447
448 auto& my_var = variances[start + r];
449 my_var = 0;
450 for (const auto& bm : block_multipliers) {
451 my_var += re_variances[bm.first][r] * bm.second;
452 }
453 my_var /= ncells - 1; // See COMMENT ON DENOMINATOR above.
454 }
455 }, ngenes, nthreads);
456 }

473template<class IrlbaSparseMatrix_, class EigenMatrix_>
474inline void project_matrix_realized_sparse(
475 const IrlbaSparseMatrix_& emat, // cell in rows, genes in the columns, CSC.
476 EigenMatrix_& components, // dims in rows, cells in columns
477 const EigenMatrix_& scaled_rotation, // genes in rows, dims in columns
478 int nthreads)
479{
480 size_t rank = scaled_rotation.cols();
481 size_t ncells = emat.rows();
482 size_t ngenes = emat.cols();
483
484 components.resize(rank, ncells); // used a transposed version for more cache efficiency.
485 components.setZero();
486
487 const auto& x = emat.get_values();
488 const auto& i = emat.get_indices();
489 const auto& p = emat.get_pointers();
490
491 if (nthreads == 1) {
492 Eigen::VectorXd multipliers(rank);
493 for (size_t g = 0; g < ngenes; ++g) {
494 multipliers.noalias() = scaled_rotation.row(g);
495 auto start = p[g], end = p[g + 1];
496 for (size_t s = start; s < end; ++s) {
497 components.col(i[s]).noalias() += x[s] * multipliers;
498 }
499 }
500
501 } else {
502 const auto& row_nonzero_starts = emat.get_secondary_nonzero_starts();
503
504#ifndef IRLBA_CUSTOM_PARALLEL
505#ifdef _OPENMP
506 #pragma omp parallel for num_threads(nthreads)
507#endif
508 for (int t = 0; t < nthreads; ++t) {
509#else
510 IRLBA_CUSTOM_PARALLEL(nthreads, [&](size_t t) -> void {
511#endif
512
513 const auto& starts = row_nonzero_starts[t];
514 const auto& ends = row_nonzero_starts[t + 1];
515 Eigen::VectorXd multipliers(rank);
516
517 for (size_t g = 0; g < ngenes; ++g) {
518 multipliers.noalias() = scaled_rotation.row(g);
519 auto start = starts[g], end = ends[g];
520 for (size_t s = start; s < end; ++s) {
521 components.col(i[s]).noalias() += x[s] * multipliers;
522 }
523 }
524
525#ifndef IRLBA_CUSTOM_PARALLEL
526 }
527#else
528 });
529#endif
530 }

533template<typename Value_, typename Index_, class EigenMatrix_>
534inline void project_matrix_transposed_tatami(
535 const tatami::Matrix<Value_, Index_>& mat, // genes in rows, cells in columns
536 EigenMatrix_& components,
537 const EigenMatrix_& scaled_rotation, // genes in rows, dims in columns
538 int nthreads)
539{
540 size_t rank = scaled_rotation.cols();
541 auto ngenes = mat.nrow(), ncells = mat.ncol();
542 typedef typename EigenMatrix_::Scalar Scalar;
543
544 components.resize(rank, ncells); // used a transposed version for more cache efficiency.
545
546 if (mat.prefer_rows()) {
547 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
548 static_assert(!EigenMatrix_::IsRowMajor);
549 auto vptr = scaled_rotation.data();
550 std::vector<Value_> vbuffer(length);
551
552 std::vector<std::vector<Scalar> > local_buffers; // create separate buffers to avoid false sharing.
553 local_buffers.reserve(rank);
554 for (size_t r = 0; r < rank; ++r) {
555 local_buffers.emplace_back(length);
556 }
557

561 for (Index_ g = 0; g < ngenes; ++g) {
562 auto range = ext->fetch(vbuffer.data(), ibuffer.data());
563 for (size_t r = 0; r < rank; ++r) {
564 auto mult = vptr[r * static_cast<size_t>(ngenes) + static_cast<size_t>(g)]; // cast to avoid overflow.
565 auto& local_buffer = local_buffers[r];
566 for (Index_ i = 0; i < range.number; ++i) {
567 local_buffer[range.index[i] - start] += range.value[i] * mult;
568 }
569 }
570 }
571
572 } else {
573 auto ext = tatami::consecutive_extractor<false>(&mat, true, static_cast<Index_>(0), ngenes, start, length);
574 for (Index_ g = 0; g < ngenes; ++g) {
575 auto ptr = ext->fetch(vbuffer.data());
576 for (size_t r = 0; r < rank; ++r) {
577 auto mult = vptr[r * static_cast<size_t>(ngenes) + static_cast<size_t>(g)]; // cast to avoid overflow.
578 auto& local_buffer = local_buffers[r];
579 for (Index_ c = 0; c < length; ++c) {
580 local_buffer[c] += ptr[c] * mult;
581 }
582 }
583 }
584 }
585
586 for (size_t r = 0; r < rank; ++r) {
587 for (Index_ c = 0; c < length; ++c) {
588 components.coeffRef(r, c + start) = local_buffers[r][c];
589 }
590 }
591
592 }, ncells, nthreads);
593
594 } else {
595 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
596 std::vector<Value_> vbuffer(ngenes);
597 static_assert(!EigenMatrix_::IsRowMajor);
598 auto optr = components.data() + static_cast<size_t>(start) * rank; // cast to avoid overflow.
599
600 if (mat.is_sparse()) {
601 std::vector<Index_> ibuffer(ngenes);
602 auto ext = tatami::consecutive_extractor<true>(&mat, false, start, length);
603
604 for (Index_ c = start, end = start + length; c < end; ++c, optr += rank) {
605 auto range = ext->fetch(vbuffer.data(), ibuffer.data());
606 static_assert(!EigenMatrix_::IsRowMajor);
607 auto vptr = scaled_rotation.data();
608 for (size_t v = 0; v < rank; ++v, vptr += ngenes) {
609 auto& output = optr[v];
610 output = 0;
611 for (Index_ i = 0; i < range.number; ++i) {
612 output += vptr[range.index[i]] * range.value[i];
613 }
614 }
615 }
616
617 } else {
618 auto ext = tatami::consecutive_extractor<false>(&mat, false, start, length);
619 for (Index_ c = start, end = start + length; c < end; ++c, optr += rank) {
620 auto ptr = ext->fetch(vbuffer.data());
621 static_assert(!EigenMatrix_::IsRowMajor);
622 auto vptr = scaled_rotation.data();
623 for (size_t v = 0; v < rank; ++v, vptr += ngenes) {
624 optr[v] = std::inner_product(vptr, vptr + ngenes, ptr, static_cast<Scalar>(0));
625 }
626 }
627 }
628 }, ncells, nthreads);

632template<class EigenMatrix_, class EigenVector_>
633void clean_up_projected(EigenMatrix_& projected, EigenVector_& D) {
634 // Empirically centering to give nice centered PCs, because we can't
635 // guarantee that the projection is centered in this manner.
636 for (size_t i = 0, iend = projected.rows(); i < iend; ++i) {
637 projected.row(i).array() -= projected.row(i).sum() / projected.cols();
638 }
639
640 // Just dividing by the number of observations - 1 regardless of weighting.
641 typename EigenMatrix_::Scalar denom = projected.cols() - 1;
642 for (auto& d : D) {
643 d = d * d / denom;
644 }
645}

647/*******************************
648 ***** Residual wrapper ********
649 *******************************/
650
651// This wrapper class mimics multiplication with the residuals,
652// i.e., after subtracting the per-block mean from each cell.
653template<class Matrix_, typename Block_, class EigenMatrix_, class EigenVector_>
654class ResidualWrapper {
655public:
656 ResidualWrapper(const Matrix_& mat, const Block_* block, const EigenMatrix_& means) : my_mat(mat), my_block(block), my_means(means) {}
657
658public:
659 Eigen::Index rows() const { return my_mat.rows(); }
660 Eigen::Index cols() const { return my_mat.cols(); }
661
662public:
663 struct Workspace {
664 Workspace(size_t nblocks, irlba::WrappedWorkspace<Matrix_> c) : sub(nblocks), child(std::move(c)) {}
665 EigenVector_ sub;
666 EigenVector_ holding;
667 irlba::WrappedWorkspace<Matrix_> child;
668 };
669
670 Workspace workspace() const {
671 return Workspace(my_means.rows(), irlba::wrapped_workspace(my_mat));
672 }
673
674 template<class Right_>
675 void multiply(const Right_& rhs, Workspace& work, EigenVector_& output) const {
676 const auto& realized_rhs = [&]() {
677 if constexpr(std::is_same<Right_, EigenVector_>::value) {
678 return rhs;
679 } else {
680 work.holding.noalias() = rhs;
681 return work.holding;
682 }
683 }();
684
685 irlba::wrapped_multiply(my_mat, realized_rhs, work.child, output);
686
687 work.sub.noalias() = my_means * realized_rhs;
688 for (Eigen::Index i = 0, end = output.size(); i < end; ++i) {
689 auto& val = output.coeffRef(i);
690 val -= work.sub.coeff(my_block[i]);
691 }
692 }

694public:

706 template<class Right_>
707 void adjoint_multiply(const Right_& rhs, AdjointWorkspace& work, EigenVector_& output) const {
708 const auto& realized_rhs = [&]() {
709 if constexpr(std::is_same<Right_, EigenVector_>::value) {
710 return rhs;
711 } else {
712 work.holding.noalias() = rhs;
713 return work.holding;
714 }
715 }();
716
717 irlba::wrapped_adjoint_multiply(my_mat, realized_rhs, work.child, output);
718
719 work.aggr.setZero();
720 for (Eigen::Index i = 0, end = realized_rhs.size(); i < end; ++i) {
721 work.aggr.coeffRef(my_block[i]) += realized_rhs.coeff(i);
722 }
723
724 output.noalias() -= my_means.adjoint() * work.aggr;
725 }

747template<bool realize_matrix_, bool sparse_, typename Value_, typename Index_, typename Block_, class EigenMatrix_, class EigenVector_>
748void run_blocked(
749 const tatami::Matrix<Value_, Index_>& mat,
750 const Block_* block,
751 const BlockingDetails<Index_, EigenVector_>& block_details,
752 int rank,
753 const BlockedPcaOptions& options,
754 EigenMatrix_& components,
755 EigenMatrix_& rotation,
756 EigenVector_& variance_explained,
757 EigenMatrix_& center_m,
758 EigenVector_& scale_v,
759 typename EigenVector_::Scalar& total_var)
760{
761 Index_ ngenes = mat.nrow(), ncells = mat.ncol();
762
763 auto emat = [&]{
764 if constexpr(!realize_matrix_) {
765 return internal::TransposedTatamiWrapper<EigenVector_, Value_, Index_>(mat, options.num_threads);
766
767 } else if constexpr(sparse_) {
768 // 'extracted' contains row-major contents... but we implicitly transpose it to CSC with genes in columns.
769 auto extracted = tatami::retrieve_compressed_sparse_contents<Value_, Index_>(&mat, /* row = */ true, /* two_pass = */ false, /* threads = */ options.num_threads);
770 return irlba::ParallelSparseMatrix(ncells, ngenes, std::move(extracted.value), std::move(extracted.index), std::move(extracted.pointers), true, options.num_threads);
771
772 } else {
773 // Perform an implicit transposition by performing a row-major extraction into a column-major transposed matrix.
774 EigenMatrix_ emat(ncells, ngenes);
775 static_assert(!EigenMatrix_::IsRowMajor);
776 tatami::convert_to_dense(&mat, /* row_major = */ true, emat.data(), options.num_threads);
777 return emat;
778 }
779 }();
780
781 auto nblocks = block_details.block_size.size();
782 center_m.resize(nblocks, ngenes);
783 scale_v.resize(ngenes);
784 if constexpr(!realize_matrix_) {
785 compute_blockwise_mean_and_variance_tatami(mat, block, block_details, center_m, scale_v, options.num_threads);
786 } else if constexpr(sparse_) {
787 compute_blockwise_mean_and_variance_realized_sparse(emat, block, block_details, center_m, scale_v, options.num_threads);
788 } else {
789 compute_blockwise_mean_and_variance_realized_dense(emat, block, block_details, center_m, scale_v, options.num_threads);
790 }
791 total_var = internal::process_scale_vector(options.scale, scale_v);
792
793 ResidualWrapper<decltype(emat), Block_, EigenMatrix_, EigenVector_> centered(emat, block, center_m);
794
795 if (block_details.weighted) {
796 if (options.scale) {
797 irlba::Scaled<true, decltype(centered), EigenVector_> scaled(centered, scale_v, /* divide = */ true);
798 irlba::Scaled<false, decltype(scaled), EigenVector_> weighted(scaled, block_details.expanded_weights, /* divide = */ false);
799 irlba::compute(weighted, rank, components, rotation, variance_explained, options.irlba_options);
800 } else {
801 irlba::Scaled<false, decltype(centered), EigenVector_> weighted(centered, block_details.expanded_weights, /* divide = */ false);
802 irlba::compute(weighted, rank, components, rotation, variance_explained, options.irlba_options);
803 }
804
805 EigenMatrix_ tmp;
806 const auto& scaled_rotation = scale_rotation_matrix(rotation, options.scale, scale_v, tmp);
807
808 // This transposes 'components' to be a NDIM * NCELLS matrix.
809 if constexpr(!realize_matrix_) {
810 project_matrix_transposed_tatami(mat, components, scaled_rotation, options.num_threads);
811 } else if constexpr(sparse_) {
812 project_matrix_realized_sparse(emat, components, scaled_rotation, options.num_threads);
813 } else {
814 components.noalias() = (emat * scaled_rotation).adjoint();
815 }
816
817 // Subtracting each block's mean from the PCs.
818 if (options.components_from_residuals) {
819 EigenMatrix_ centering = (center_m * scaled_rotation).adjoint();
820 for (size_t i = 0, iend = components.cols(); i < iend; ++i) {
821 components.col(i) -= centering.col(block[i]);
822 }
823 }
824
825 clean_up_projected(components, variance_explained);
826 if (!options.transpose) {
827 components.adjointInPlace();
828 }
829
830 } else {
831 if (options.scale) {
832 irlba::Scaled<true, decltype(centered), EigenVector_> scaled(centered, scale_v, /* divide = */ true);
833 irlba::compute(scaled, rank, components, rotation, variance_explained, options.irlba_options);
834 } else {
835 irlba::compute(centered, rank, components, rotation, variance_explained, options.irlba_options);
836 }
837
838 if (options.components_from_residuals) {
839 internal::clean_up(mat.ncol(), components, variance_explained);
840 if (options.transpose) {
841 components.adjointInPlace();
842 }
843 } else {
844 EigenMatrix_ tmp;
845 const auto& scaled_rotation = scale_rotation_matrix(rotation, options.scale, scale_v, tmp);
846
847 // This transposes 'components' to be a NDIM * NCELLS matrix.
848 if constexpr(!realize_matrix_) {
849 project_matrix_transposed_tatami(mat, components, scaled_rotation, options.num_threads);
850 } else if constexpr(sparse_) {
851 project_matrix_realized_sparse(emat, components, scaled_rotation, options.num_threads);
852 } else {
853 components.noalias() = (emat * scaled_rotation).adjoint();
854 }
855
856 clean_up_projected(components, variance_explained);
857 if (!options.transpose) {
858 components.adjointInPlace();
859 }
860 }
861 }
862}
