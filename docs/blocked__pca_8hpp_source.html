103template<typename Index_, class EigenVector_>
104struct BlockingDetails {
105 std::vector<Index_> block_size;
106
107 bool weighted = false;
108 typedef typename EigenVector_::Scalar Weight;
109
110 // The below should only be used if weighted = true.
111 std::vector<Weight> per_element_weight;
112 Weight total_block_weight = 0;
113 EigenVector_ expanded_weights;
114};
115
116template<class EigenVector_, typename Index_, typename Block_>
117BlockingDetails<Index_, EigenVector_> compute_blocking_details(
118 Index_ ncells,
119 const Block_* block,
120 scran_blocks::WeightPolicy block_weight_policy,
121 const scran_blocks::VariableWeightParameters& variable_block_weight_parameters)
122{
123 BlockingDetails<Index_, EigenVector_> output;
124 output.block_size = tatami_stats::tabulate_groups(block, ncells);
125 if (block_weight_policy == scran_blocks::WeightPolicy::NONE) {
126 return output;
127 }
128
129 const auto& block_size = output.block_size;
130 size_t nblocks = block_size.size();
131 output.weighted = true;
132 auto& total_weight = output.total_block_weight;
133 auto& element_weight = output.per_element_weight;
134 element_weight.resize(nblocks);
135
136 for (size_t i = 0; i < nblocks; ++i) {
137 auto bsize = block_size[i];
138
139 // Computing effective block weights that also incorporate division by the
140 // block size. This avoids having to do the division by block size in the
141 // 'compute_blockwise_mean_and_variance*()' functions.
142 if (bsize) {
143 typename EigenVector_::Scalar block_weight = 1;
144 if (block_weight_policy == scran_blocks::WeightPolicy::VARIABLE) {
145 block_weight = scran_blocks::compute_variable_weight(bsize, variable_block_weight_parameters);
146 }
147
148 element_weight[i] = block_weight / bsize;
149 total_weight += block_weight;
150 } else {
151 element_weight[i] = 0;
152 }
153 }
154
155 // Setting a placeholder value to avoid problems with division by zero.
156 if (total_weight == 0) {
157 total_weight = 1;
158 }
159
160 // Expanding them for multiplication in the IRLBA wrappers.
161 auto sqrt_weights = element_weight;
162 for (auto& s : sqrt_weights) {
163 s = std::sqrt(s);
164 }
165
166 auto& expanded = output.expanded_weights;
167 expanded.resize(ncells);
168 for (Index_ i = 0; i < ncells; ++i) {
169 expanded.coeffRef(i) = sqrt_weights[block[i]];
170 }
171
172 return output;
173}

179template<typename Num_, typename Value_, typename Index_, typename Block_, typename EigenVector_, typename Float_>
180void compute_sparse_mean_and_variance_blocked(
181 Num_ num_nonzero,
182 const Value_* values,
183 const Index_* indices,
184 const Block_* block,
185 const BlockingDetails<Index_, EigenVector_>& block_details,
186 Float_* centers,
187 Float_& variance,
188 std::vector<Index_>& block_copy,
189 Num_ num_all)
190{
191 const auto& block_size = block_details.block_size;
192 size_t nblocks = block_size.size();
193
194 std::fill_n(centers, nblocks, 0);
195 for (Num_ i = 0; i < num_nonzero; ++i) {
196 centers[block[indices[i]]] += values[i];
197 }
198 for (size_t b = 0; b < nblocks; ++b) {
199 auto bsize = block_size[b];
200 if (bsize) {
201 centers[b] /= bsize;
202 }
203 }
204
205 // Computing the variance from the sum of squared differences.
206 // This is technically not the correct variance estimate if we
207 // were to consider the loss of residual d.f. from estimating
208 // the block means, but it's what the PCA sees, so whatever.
209 variance = 0;
210 std::copy(block_size.begin(), block_size.end(), block_copy.begin());
211
212 if (block_details.weighted) {
213 for (Num_ i = 0; i < num_nonzero; ++i) {
214 Block_ curb = block[indices[i]];
215 auto diff = values[i] - centers[curb];
216 variance += diff * diff * block_details.per_element_weight[curb];
217 --block_copy[curb];
218 }
219 for (size_t b = 0; b < nblocks; ++b) {
220 auto val = centers[b];
221 variance += val * val * block_copy[b] * block_details.per_element_weight[b];
222 }
223 } else {
224 for (Num_ i = 0; i < num_nonzero; ++i) {
225 Block_ curb = block[indices[i]];
226 auto diff = values[i] - centers[curb];
227 variance += diff * diff;
228 --block_copy[curb];
229 }
230 for (size_t b = 0; b < nblocks; ++b) {
231 auto val = centers[b];
232 variance += val * val * block_copy[b];
233 }
234 }
235
236 // COMMENT ON DENOMINATOR:
237 // If we're not dealing with weights, we compute the actual sample
238 // variance for easy interpretation (and to match up with the
239 // per-PC calculations in internal::clean_up).
240 //
241 // If we're dealing with weights, the concept of the sample variance
242 // becomes somewhat weird, but we just use the same denominator for
243 // consistency in clean_up_projected. Magnitude doesn't matter when
244 // scaling for internal::process_scale_vector anyway.
245 variance /= num_all - 1;
246}

285template<typename Num_, typename Value_, typename Block_, typename Index_, typename EigenVector_, typename Float_>
286void compute_dense_mean_and_variance_blocked(
287 Num_ number,
288 const Value_* values,
289 const Block_* block,
290 const BlockingDetails<Index_, EigenVector_>& block_details,
291 Float_* centers,
292 Float_& variance)
293{
294 const auto& block_size = block_details.block_size;
295 size_t nblocks = block_size.size();
296 std::fill_n(centers, nblocks, 0);
297 for (Num_ r = 0; r < number; ++r) {
298 centers[block[r]] += values[r];
299 }
300 for (size_t b = 0; b < nblocks; ++b) {
301 const auto& bsize = block_size[b];
302 if (bsize) {
303 centers[b] /= bsize;
304 }
305 }
306
307 variance = 0;
308
309 if (block_details.weighted) {
310 for (Num_ r = 0; r < number; ++r) {
311 auto curb = block[r];
312 auto delta = values[r] - centers[curb];
313 variance += delta * delta * block_details.per_element_weight[curb];
314 }
315 } else {
316 for (Num_ r = 0; r < number; ++r) {
317 auto curb = block[r];
318 auto delta = values[r] - centers[curb];
319 variance += delta * delta;
320 }
321 }
322
323 variance /= number - 1; // See COMMENT ON DENOMINATOR above.
324}

350template<typename Value_, typename Index_, typename Block_, class EigenMatrix_, class EigenVector_>
351void compute_blockwise_mean_and_variance_tatami(
352 const tatami::Matrix<Value_, Index_>& mat, // this should have genes in the rows!
353 const Block_* block,
354 const BlockingDetails<Index_, EigenVector_>& block_details,
355 EigenMatrix_& centers,
356 EigenVector_& variances,
357 int nthreads)
358{
359 const auto& block_size = block_details.block_size;
360 size_t nblocks = block_size.size();
361 Index_ ngenes = mat.nrow();
362 Index_ ncells = mat.ncol();
363
364 if (mat.prefer_rows()) {
365 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
366 static_assert(!EigenMatrix_::IsRowMajor);
367 auto mptr = centers.data() + static_cast<size_t>(start) * nblocks; // cast to avoid overflow.
368 std::vector<Index_> block_copy(nblocks);
369
370 std::vector<Value_> vbuffer(ncells);
371
372 if (mat.is_sparse()) {
373 std::vector<Index_> ibuffer(ncells);
374 auto ext = tatami::consecutive_extractor<true>(&mat, true, start, length);
375 for (Index_ r = start, end = start + length; r < end; ++r, mptr += nblocks) {
376 auto range = ext->fetch(vbuffer.data(), ibuffer.data());
377 compute_sparse_mean_and_variance_blocked(range.number, range.value, range.index, block, block_details, mptr, variances[r], block_copy, ncells);
378 }
379 } else {
380 auto ext = tatami::consecutive_extractor<false>(&mat, true, start, length);
381 for (Index_ r = start, end = start + length; r < end; ++r, mptr += nblocks) {
382 auto ptr = ext->fetch(vbuffer.data());
383 compute_dense_mean_and_variance_blocked(ncells, ptr, block, block_details, mptr, variances[r]);
384 }
385 }
386 }, ngenes, nthreads);
387
388 } else {
389 typedef typename EigenVector_::Scalar Scalar;
390
391 std::vector<std::pair<size_t, Scalar> > block_multipliers;
392 block_multipliers.reserve(nblocks);
393 for (size_t b = 0; b < nblocks; ++b) {
394 auto bsize = block_size[b];
395 if (bsize > 1) { // skipping blocks with NaN variances.
396 Scalar mult = bsize - 1; // need to convert variances back into sum of squared differences.
397 if (block_details.weighted) {
398 mult *= block_details.per_element_weight[b];
399 }
400 block_multipliers.emplace_back(b, mult);
401 }
402 }
403
404 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
405 std::vector<std::vector<Scalar> > re_centers, re_variances;
406 re_centers.reserve(nblocks);
407 re_variances.reserve(nblocks);
408 for (size_t b = 0; b < nblocks; ++b) {
409 re_centers.emplace_back(length);
410 re_variances.emplace_back(length);
411 }
412
413 std::vector<Value_> vbuffer(length);
414
415 if (mat.is_sparse()) {
416 std::vector<tatami_stats::variances::RunningSparse<Scalar, Value_, Index_> > running;
417 running.reserve(nblocks);
418 for (size_t b = 0; b < nblocks; ++b) {
419 running.emplace_back(length, re_centers[b].data(), re_variances[b].data(), /* skip_nan = */ false, /* subtract = */ start);
420 }
421
422 std::vector<Index_> ibuffer(length);
423 auto ext = tatami::consecutive_extractor<true>(&mat, false, static_cast<Index_>(0), ncells, start, length);
424 for (Index_ c = 0; c < ncells; ++c) {
425 auto range = ext->fetch(vbuffer.data(), ibuffer.data());
426 running[block[c]].add(range.value, range.index, range.number);
427 }
428
429 for (size_t b = 0; b < nblocks; ++b) {
430 running[b].finish();
431 }
432
433 } else {
434 std::vector<tatami_stats::variances::RunningDense<Scalar, Value_, Index_> > running;
435 running.reserve(nblocks);
436 for (size_t b = 0; b < nblocks; ++b) {
437 running.emplace_back(length, re_centers[b].data(), re_variances[b].data(), /* skip_nan = */ false);
438 }
439
440 auto ext = tatami::consecutive_extractor<false>(&mat, false, static_cast<Index_>(0), ncells, start, length);
441 for (Index_ c = 0; c < ncells; ++c) {
442 auto ptr = ext->fetch(vbuffer.data());
443 running[block[c]].add(ptr);
444 }
445
446 for (size_t b = 0; b < nblocks; ++b) {
447 running[b].finish();
448 }
449 }
450
451 static_assert(!EigenMatrix_::IsRowMajor);
452 auto mptr = centers.data() + static_cast<size_t>(start) * nblocks; // cast to avoid overflow.
453 for (Index_ r = 0; r < length; ++r, mptr += nblocks) {
454 for (size_t b = 0; b < nblocks; ++b) {
455 mptr[b] = re_centers[b][r];
456 }
457
458 auto& my_var = variances[start + r];
459 my_var = 0;
460 for (const auto& bm : block_multipliers) {
461 my_var += re_variances[bm.first][r] * bm.second;
462 }
463 my_var /= ncells - 1; // See COMMENT ON DENOMINATOR above.
464 }
465 }, ngenes, nthreads);
466 }

483template<class IrlbaSparseMatrix_, class EigenMatrix_>
484inline void project_matrix_realized_sparse(
485 const IrlbaSparseMatrix_& emat, // cell in rows, genes in the columns, CSC.
486 EigenMatrix_& components, // dims in rows, cells in columns
487 const EigenMatrix_& scaled_rotation, // genes in rows, dims in columns
488 int nthreads)
489{
490 size_t rank = scaled_rotation.cols();
491 size_t ncells = emat.rows();
492 size_t ngenes = emat.cols();
493
494 components.resize(rank, ncells); // used a transposed version for more cache efficiency.
495 components.setZero();
496
497 const auto& x = emat.get_values();
498 const auto& i = emat.get_indices();
499 const auto& p = emat.get_pointers();
500
501 if (nthreads == 1) {
502 Eigen::VectorXd multipliers(rank);
503 for (size_t g = 0; g < ngenes; ++g) {
504 multipliers.noalias() = scaled_rotation.row(g);
505 auto start = p[g], end = p[g + 1];
506 for (size_t s = start; s < end; ++s) {
507 components.col(i[s]).noalias() += x[s] * multipliers;
508 }
509 }
510
511 } else {
512 const auto& row_nonzero_starts = emat.get_secondary_nonzero_starts();
513
514 irlba::parallelize(nthreads, [&](size_t t) -> void {
515 const auto& starts = row_nonzero_starts[t];
516 const auto& ends = row_nonzero_starts[t + 1];
517 Eigen::VectorXd multipliers(rank);
518
519 for (size_t g = 0; g < ngenes; ++g) {
520 multipliers.noalias() = scaled_rotation.row(g);
521 auto start = starts[g], end = ends[g];
522 for (size_t s = start; s < end; ++s) {
523 components.col(i[s]).noalias() += x[s] * multipliers;
524 }
525 }
526 });
527 }

602

607 for (Index_ c = start, end = start + length; c < end; ++c, optr += rank) {
608 auto range = ext->fetch(vbuffer.data(), ibuffer.data());

618 }

635template<class EigenMatrix_, class EigenVector_>
636void clean_up_projected(EigenMatrix_& projected, EigenVector_& D) {
637 // Empirically centering to give nice centered PCs, because we can't
638 // guarantee that the projection is centered in this manner.
639 for (size_t i = 0, iend = projected.rows(); i < iend; ++i) {
640 projected.row(i).array() -= projected.row(i).sum() / projected.cols();
641 }
642
643 // Just dividing by the number of observations - 1 regardless of weighting.
644 typename EigenMatrix_::Scalar denom = projected.cols() - 1;
645 for (auto& d : D) {
646 d = d * d / denom;
647 }
648}

650/*******************************
651 ***** Residual wrapper ********
652 *******************************/
653
654// This wrapper class mimics multiplication with the residuals,
655// i.e., after subtracting the per-block mean from each cell.
656template<class Matrix_, typename Block_, class EigenMatrix_, class EigenVector_>
657class ResidualWrapper {
658public:
659 ResidualWrapper(const Matrix_& mat, const Block_* block, const EigenMatrix_& means) : my_mat(mat), my_block(block), my_means(means) {}
660
661public:
662 Eigen::Index rows() const { return my_mat.rows(); }
663 Eigen::Index cols() const { return my_mat.cols(); }
664
665public:
666 struct Workspace {
667 Workspace(size_t nblocks, irlba::WrappedWorkspace<Matrix_> c) : sub(nblocks), child(std::move(c)) {}
668 EigenVector_ sub;
669 EigenVector_ holding;
670 irlba::WrappedWorkspace<Matrix_> child;
671 };
672
673 Workspace workspace() const {
674 return Workspace(my_means.rows(), irlba::wrapped_workspace(my_mat));
675 }
676
677 template<class Right_>
678 void multiply(const Right_& rhs, Workspace& work, EigenVector_& output) const {
679 const auto& realized_rhs = [&]() {
680 if constexpr(std::is_same<Right_, EigenVector_>::value) {
681 return rhs;
682 } else {
683 work.holding.noalias() = rhs;
684 return work.holding;
685 }
686 }();
687
688 irlba::wrapped_multiply(my_mat, realized_rhs, work.child, output);
689
690 work.sub.noalias() = my_means * realized_rhs;
691 for (Eigen::Index i = 0, end = output.size(); i < end; ++i) {
692 auto& val = output.coeffRef(i);
693 val -= work.sub.coeff(my_block[i]);
694 }
695 }
696
697public:
698 struct AdjointWorkspace {

701 EigenVector_ holding;
702 irlba::WrappedAdjointWorkspace<Matrix_> child;

709 template<class Right_>
710 void adjoint_multiply(const Right_& rhs, AdjointWorkspace& work, EigenVector_& output) const {
711 const auto& realized_rhs = [&]() {
712 if constexpr(std::is_same<Right_, EigenVector_>::value) {
713 return rhs;
714 } else {
715 work.holding.noalias() = rhs;
716 return work.holding;
717 }
718 }();
719
720 irlba::wrapped_adjoint_multiply(my_mat, realized_rhs, work.child, output);
721
722 work.aggr.setZero();
723 for (Eigen::Index i = 0, end = realized_rhs.size(); i < end; ++i) {
724 work.aggr.coeffRef(my_block[i]) += realized_rhs.coeff(i);
725 }
726
727 output.noalias() -= my_means.adjoint() * work.aggr;
728 }

732 EigenMatrix2_ realize() const {
733 EigenMatrix2_ output = irlba::wrapped_realize<EigenMatrix2_>(my_mat);
734 for (Eigen::Index r = 0, rend = output.rows(); r < rend; ++r) {
735 output.row(r) -= my_means.row(my_block[r]);
736 }
737 return output;
738 }

743 const EigenMatrix_& my_means;

750template<bool realize_matrix_, bool sparse_, typename Value_, typename Index_, typename Block_, class EigenMatrix_, class EigenVector_>
751void run_blocked(
752 const tatami::Matrix<Value_, Index_>& mat,
753 const Block_* block,
754 const BlockingDetails<Index_, EigenVector_>& block_details,
755 const BlockedPcaOptions& options,
756 EigenMatrix_& components,
757 EigenMatrix_& rotation,
758 EigenVector_& variance_explained,
759 EigenMatrix_& center_m,
760 EigenVector_& scale_v,
761 typename EigenVector_::Scalar& total_var,
762 bool& converged)
763{
764 Index_ ngenes = mat.nrow(), ncells = mat.ncol();
765
766 auto emat = [&]{
767 if constexpr(!realize_matrix_) {
768 return internal::TransposedTatamiWrapper<EigenVector_, Value_, Index_>(mat, options.num_threads);
769
770 } else if constexpr(sparse_) {
771 // 'extracted' contains row-major contents... but we implicitly transpose it to CSC with genes in columns.
772 auto extracted = tatami::retrieve_compressed_sparse_contents<Value_, Index_>(&mat, /* row = */ true, /* two_pass = */ false, /* threads = */ options.num_threads);
773 return irlba::ParallelSparseMatrix(ncells, ngenes, std::move(extracted.value), std::move(extracted.index), std::move(extracted.pointers), true, options.num_threads);
774
775 } else {
776 // Perform an implicit transposition by performing a row-major extraction into a column-major transposed matrix.
777 EigenMatrix_ emat(ncells, ngenes);
778 static_assert(!EigenMatrix_::IsRowMajor);
779 tatami::convert_to_dense(&mat, /* row_major = */ true, emat.data(), options.num_threads);
780 return emat;
781 }
782 }();
783
784 auto nblocks = block_details.block_size.size();
785 center_m.resize(nblocks, ngenes);
786 scale_v.resize(ngenes);
787 if constexpr(!realize_matrix_) {
788 compute_blockwise_mean_and_variance_tatami(mat, block, block_details, center_m, scale_v, options.num_threads);
789 } else if constexpr(sparse_) {
790 compute_blockwise_mean_and_variance_realized_sparse(emat, block, block_details, center_m, scale_v, options.num_threads);
791 } else {
792 compute_blockwise_mean_and_variance_realized_dense(emat, block, block_details, center_m, scale_v, options.num_threads);
793 }
794 total_var = internal::process_scale_vector(options.scale, scale_v);
795
796 ResidualWrapper<decltype(emat), Block_, EigenMatrix_, EigenVector_> centered(emat, block, center_m);
797
798 if (block_details.weighted) {
799 if (options.scale) {
800 irlba::Scaled<true, decltype(centered), EigenVector_> scaled(centered, scale_v, /* divide = */ true);
801 irlba::Scaled<false, decltype(scaled), EigenVector_> weighted(scaled, block_details.expanded_weights, /* divide = */ false);
802 auto out = irlba::compute(weighted, options.number, components, rotation, variance_explained, options.irlba_options);
803 converged = out.first;
804 } else {
805 irlba::Scaled<false, decltype(centered), EigenVector_> weighted(centered, block_details.expanded_weights, /* divide = */ false);
806 auto out = irlba::compute(weighted, options.number, components, rotation, variance_explained, options.irlba_options);
807 converged = out.first;
808 }
809
810 EigenMatrix_ tmp;
811 const auto& scaled_rotation = scale_rotation_matrix(rotation, options.scale, scale_v, tmp);
812
813 // This transposes 'components' to be a NDIM * NCELLS matrix.
814 if constexpr(!realize_matrix_) {
815 project_matrix_transposed_tatami(mat, components, scaled_rotation, options.num_threads);
816 } else if constexpr(sparse_) {
817 project_matrix_realized_sparse(emat, components, scaled_rotation, options.num_threads);
818 } else {
819 components.noalias() = (emat * scaled_rotation).adjoint();
820 }
821
822 // Subtracting each block's mean from the PCs.
823 if (options.components_from_residuals) {
824 EigenMatrix_ centering = (center_m * scaled_rotation).adjoint();
825 for (size_t i = 0, iend = components.cols(); i < iend; ++i) {
826 components.col(i) -= centering.col(block[i]);
827 }
828 }
829
830 clean_up_projected(components, variance_explained);
831 if (!options.transpose) {
832 components.adjointInPlace();
833 }
834
835 } else {
836 if (options.scale) {
837 irlba::Scaled<true, decltype(centered), EigenVector_> scaled(centered, scale_v, /* divide = */ true);
838 auto out = irlba::compute(scaled, options.number, components, rotation, variance_explained, options.irlba_options);
839 converged = out.first;
840 } else {
841 auto out = irlba::compute(centered, options.number, components, rotation, variance_explained, options.irlba_options);
842 converged = out.first;
843 }
844
845 if (options.components_from_residuals) {
846 internal::clean_up(mat.ncol(), components, variance_explained);
847 if (options.transpose) {
848 components.adjointInPlace();
849 }
850 } else {
851 EigenMatrix_ tmp;
852 const auto& scaled_rotation = scale_rotation_matrix(rotation, options.scale, scale_v, tmp);
853
854 // This transposes 'components' to be a NDIM * NCELLS matrix.
855 if constexpr(!realize_matrix_) {
856 project_matrix_transposed_tatami(mat, components, scaled_rotation, options.num_threads);
857 } else if constexpr(sparse_) {
858 project_matrix_realized_sparse(emat, components, scaled_rotation, options.num_threads);
859 } else {
860 components.noalias() = (emat * scaled_rotation).adjoint();
861 }
862
863 clean_up_projected(components, variance_explained);
864 if (!options.transpose) {
865 components.adjointInPlace();
866 }
867 }
868 }
869}
